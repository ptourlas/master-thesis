\chapter{Εννοιολογικές Τεχνολογίες}
\label{SemTech}

\quad Οι Εννοιολογικές Τεχνολογίες αποτελούνται από μια σειρά εργαλείων που διευρύνουν
τις δυνατότητες του Παγκόσμιου Ιστού (\eng{World Wide Web}).
Μέσα από τη χρήση τους επιδιώκεται η σύνδεση
διαφορετικών ειδών δεδομένων με βάση τις έννοιες που περιγράφουν. Για να καταλάβουμε
τι σημαίνει αυτό, θα πρέπει αρχικά να κατανοήσουμε τα δομικά μέρη του Παγκόσμιου
Ιστού.

\section{Ο Παγκόσμιος Ιστός ως ιστός αρχείων}

\quad Ο Παγκόσμιος Ιστός (Π.Ι. εν συντομία) είναι ένα εξαιρετικά επιτυχημένο εργαλείο
που βασίστηκε σε μια απλή ιδέα: την πρόσβαση σε αρχεία που περιέχουν συνδέσμους σε
άλλα αρχεία και επιτρέπουν την απευθείας μετάβαση σε αυτά. Η ιδέα αυτή, γνωστή ως
\emph{υπερκείμενο}\footnote{\eng{https://en.wikipedia.org/wiki/Hypertext\#History}}
ήταν ήδη γνωστή από την δεκαετία του 1960 αλλά οι πραγματικές της δυνατότητες 
αξιοποιήθηκαν το 1989 από τον \eng{Sir Tim Berners-Lee} με τη δημιουργία της πρώτης
ιστοσελίδας\footnote{\eng{http://info.cern.ch/hypertext/WWW/TheProject.html}}
του Π.Ι. Ένα λεπτό σημείο που θα πρέπει να αναφερθεί εδώ είναι η 
διάκριση μεταξύ Διαδικτύου και Π.Ι.:
το Διαδίκτυο αποτελεί ένα παγκόσμιο δίκτυο συσκευών και υποδικτύων που 
βασίζονται στα πρωτόκολλα 
\eng{TCP/IP}\footnote{\eng{https://datatracker.ietf.org/doc/html/rfc1180}} 
για την μεταξύ τους επικοινωνία ενώ
ο Π.Ι. συνδυάζει την υποδομή του Διαδικτύου με μια σειρά από άλλες τεχνολογίες
διασύνδεσης:

\begin{itemize}
	\item το Σύστημα Ονοματοδοσίας Διαδικτύου (γνωστό και ως \eng{Domain Name System}
	ή \eng{DNS}\footnote{\eng{https://datatracker.ietf.org/doc/html/rfc1034}})
	για τον συσχετισμό των \eng{domain names} με τις διευθύνσεις ΙΡ.

	\item το πρωτόκολλο \eng{HTTP\footnote{https://www.ietf.org/rfc/rfc2616.html}
	(HyperText Transfer Protocol)} το οποίο καθορίζει
	τον τρόπο επικοινωνίας μεταξύ ενός διακομιστή και ενός πελάτη στον Π.Ι.

	\item τους Ενιαίους Εντοπιστές Πόρων (γνωστούς και ως \eng{Uniform Resource Locators}
	ή \eng{URLs\footnote{https://url.spec.whatwg.org/\#concept-url}}
	), που επιτρέπουν την ταυτοποίηση αρχείων με μοναδικό τρόπο στον Π.Ι.
\end{itemize}

\begin{remark}
	\label{URIRemark}
	Τα \eng{URLs} αποτελούν υποσύνολο των Ενιαίων Αναγνωριστικών Πόρων 
	(γνωστοί και ως 
	\eng{Uniform Resource Identifiers} ή
	% for fuck's sake ALWAYS MAKE SURE YOU ESCAPE MATH MODE STUFF LIKE _
	\eng{URIs \footnote{https://www.w3.org/Addressing/URL/URI\_Overview.html}}).
	Τα \eng{URIs} είναι ένα σημαντικό εργαλείο των Εννοιολογικών Τεχνολογιών
	καθώς χρησιμοποιούνται για να περιγράψουν όχι μόνο τοποθεσίες ενός δικτύου
	αλλά και τοποθεσίες του πραγματικού κόσμου, ανθρώπους, αντικείμενα ή ακόμα
	και αφηρημένες έννοιες.
\end{remark}

\section{Πληροφορία και γνώση στον Παγκόσμιο Ιστό}

\quad Ο Π.Ι. ουσιαστικά υλοποιεί ένα κανάλι μεταφοράς δεδομένων μεταξύ ανθρώπων.
Εννοούμε δηλαδή ότι ο δημιουργός μιας σελίδας τη "δημοσιεύει" σε μια τοποθεσία
την οποία μπορεί να επισκεφθεί οποιοσδήποτε τρίτος έχει στη διάθεσή του έναν
απλό φυλλομετρητή (ή αλλιώς \eng{web browser}). Αυτό το μοντέλο επικοινωνίας
ονομάζεται \emph{αρχιτεκτονική πελάτη/διακομιστή}
\footnote{\eng{https://en.wikipedia.org/wiki/Client-server\_model}}.
Εδώ ο ρόλος των τεχνολογικών υποδομών είναι να καταστήσουν δυνατή την πρόσβαση σε
μια τοποθεσία και να παρουσιάσουν τα δεδομένα της. Πέρα από αυτό, κάθε διαδικασία
κατανόησης και παραγωγής γνώσης γίνεται από τον άνθρωπο-αποδέκτη
(βλ. Σχήμα \ref{fig:clientserver}).

\begin{figure}[ht]
	\includegraphics[scale=0.5]{thesisfig/client-server.eps}
	\caption{Η παραγωγή γνώσης από δεδομένα στον Π.Ι.}
	\label{fig:clientserver}
\end{figure}

Με παρόμοιο τρόπο θα μπορούσαμε να πούμε ότι δουλεύει και ένας πίνακας
ανακοινώσεων στον πραγματικό κόσμο.

\begin{example}
	\label{guitarboard}
	Έστω ότι κάποιος αναρτά μια αγγελία πώλησης μιας ηλεκτρικής κιθάρας στον πίνακα
	ανακοινώσεων του διαδρόμου ενός πανεπιστημίου. Τότε η
	αγγελία αυτή είναι αυτομάτως διαθέσιμη σε οποιονδήποτε γνωρίζει την τοποθεσία
	του πίνακα στο πανεπιστήμιο και μπορεί να διαβάσει τη γλώσσα στην οποία είναι
	γραμμένη η αγγελία.
\end{example}

Στο παράδειγμα \ref{guitarboard} κάνουμε μια έμμεση παραδοχή: αγνοούμε την ανάγκη
να δώσουμε μια περιγραφή της κιθάρας σαν μουσικό όργανο καθώς τη θεωρούμε κοινή
γνώση. Τι θα γινόταν όμως αν στη θέση της κιθάρας είχαμε ένα
\eng{didgeridoo\footnote{https://en.wikipedia.org/wiki/Didgeridoo}}?

Σίγουρα θα δυσκόλευε την πώληση το γεγονός ότι είναι λίγοι οι υποψήφιοι
αγοραστές που πιθανώς να γνωρίζουν κάποια πράγματα για αυτό το εξωτικό μουσικό
όργανο. Θα μπορούσαμε όμως να βελτιώσουμε την αγγελία μας συνοδεύοντας την και
από μια σελίδα με ιστορικές πληροφορίες και το πως μπορεί να παίξει κανείς
μουσική με ένα \eng{didgeridoo}. Τότε ο αναγνώστης της αγγελίας μας αποκτά
μια σφαιρικότερη είκονα του τι είναι αυτό το μουσικό όργανο. Κατά κάποιο τρόπο,
μπορούμε να πούμε ότι η αγγελία μας δεν μεταφέρει απλώς δεδομένα όπως το όνομα
ενός οργάνου και την τιμή του αλλά εσωκλείει και "γνώση" σχετική με το αντικείμενο
που μας ενδιαφέρει.

Επιστρέφοντας στον κόσμο του Διαδικτύου, τον ίδιο στόχο πετυχαίνουμε με τις
Εννοιολογικές Τεχνολογίες. Εμπλουτίζουμε τα διαθέσιμα δεδομένα κατάλληλα ώστε να
παρουσιάζουν ευρύτερες έννοιες. Μια μεγάλη διαφορά όμως είναι μια επιπλέον
επιδίωξή μας: Τα δεδομένα δεν θέλουμε απλά να παρουσιάζουν και έννοιες, θέλουμε
ταυτόχρονα να είναι επεξεργάσιμα από μηχανές. Έτσι η διαδικασία παραγωγής γνώσης
μπορεί, σε κάποιο βαθμό, να αυτοματοποιηθεί.

\begin{remark}
	Μιλάμε για \underline{μερική} αυτοματοποίηση της παραγωγής γνώσης για
	τους εξής λόγους:

	\begin{itemize}
		\item Η φυσική γλώσσα έχει τόσο μεγάλες εκφραστικές δυνατότητες που
		μπορούν να οδηγήσουν σε παράδοξα και σε μη-αποφάνσιμα
		προβλήματα.\footnote{
			Αποφάνσιμο (\eng{decidable}) ονομάζεται ένα πρόβλημα για το οποίο
			υπάρχει αλγόριθμος που οδηγεί πάντα σε μια σωστή απάντηση, είτε 
			θετική ή αρνητική.
		}
		’ρα μια απόπειρα να κατανοήσουμε πλήρως αλγοριθμικά την φυσική γλώσσα είναι
		εκ προοιμίου καταδικασμένη.

		\item Στη φυσική γλώσσα υπάρχει (συνήθως) μια κοινή γνώση μεταξύ των
		συνομιλητών, η οποία και καθορίζει τα συμφραζόμενα. Συμπληρώνουμε δηλαδή
		τα νοηματικά κενά των πληροφοριών που δεχόμαστε έτσι ώστε να μπορούμε να 
		τις κατανοήσουμε. Η διαδικασία αυτή συμβαίνει στο μυαλό του κάθε συνομιλητή
		δυναμικά και μπορεί να οδηγήσει σε διαφορετική κατανόηση.
	\end{itemize}
\end{remark}

Οι παραπάνω παράγοντες οδηγούν στην ανάγκη για κοινά εργαλεία και πρότυπα 
προκειμένου να υλοποιήσουμε τις Εννοιολογικές Τεχνολογίες. Ένα πρώτο πεδίο εφαρμογής
που θα εξετάσουμε είναι ο Εννοιολογικός Ιστός.

\begin{remark}
	Υπάρχει διάκριση ανάμεσα στους όρους "Εννοιολογικές Tεχνολογίες" και
	"Εννοιολογικός Ιστός". Ο Εννοιολογικός Ιστός δεν αποτελεί την μοναδική
	εφαρμογή αυτών των τεχνολογιών. ’λλα παραδείγματα εφαρμογών συναντώνται:
	
	\begin{itemize}
		\item στη Bιομηχανία, όπου οι Εννοιολογικές
		Τεχνολογίες μπορούν να χρησιμοποιηθούν για να καθοριστεί ένα πεδίο γνώσης
		με σαφή τρόπο για όλους τους ενδιαφερόμενους
		
		\item στις Εννοιολογικές Βάσεις
		Δεδομένων\footnote{\eng{https://skr3.nlm.nih.gov/SemMedDB/}}, οι οποίες
		επιτρέπουν την αποθήκευση των σχέσεων που συνδέουν τα δεδομένα και τη 
		σύνθεση νέων σχέσεων για την εξαγωγή γνώσης.
	\end{itemize} 
\end{remark}

\section{Αρχιτεκτονική του Εννοιολογικού Ιστού}

\quad Ο Εννοιολογικός Ιστός αποτελεί μια πολυεπίπεδη αρχιτεκτονική. Ξεκινώντας από τα
χαμηλότερα στρώματα που καθορίζουν τους συντακτικούς κανόνες και την κατάλληλη
κωδικοποίηση (\eng{UTF-8, URI, XML}),
περνάμε σε αυτά που καθορίζουν την ανταλλαγή δεδομένων (\eng{RDF}) και την βασική
σημασιολογική τους ταξινόμηση (\eng{RDFS}). Έτσι έχουμε μια υποδομή πάνω στην
οποία μπορούμε να επεκτείνουμε την σημασιολογική ταξινόμηση με τρόπο που να επιτρέπει
τις αλγοριθμικές εφαρμογές (\eng{OWL}), να ορίζουμε σύνολα κανόνων (\eng{RIF})
και να αναζητούμε πληροφορίες (\eng{SPARQL}). Στο τέλος τοποθετούμε
εργαλεία που ενοποιούν τα κατώτερα στρώματα, μας παρέχουν τη δυνατότητα να
πιστοποιούμε τις πηγές των δεδομένων και καθορίζουν τον βαθμό αξιοπιστίας τους.
Πρέπει να σημειωθεί επιπλέον και η χρήση εργαλείων της Κρυπτογραφίας όπως το
πρωτόκολλο \eng{HTTPS} και οι ψηφιακές υπογραφές προκειμένου να εξασφαλίζεται
η ασφαλής επικοινωνία και η ακεραιότητα των δεδομένων.

Η σύνθεση της αρχιτεκτονικής που περιγράψαμε φαίνεται στο σχήμα
\ref{fig:semanticlayers}. Στη συνέχεια θα δώσουμε μια συνοπτική περιγραφή
για κάθε ένα από τα στρώματα που αναφέραμε καθώς μια ανάλυση των δυσκολιών
που συναντώνται στην ενοποίησή τους.

\begin{figure}[ht]

\subsection{Κωδικοποίηση κειμένου και συντακτικοί κανόνες}

\quad Η κωδικοποίηση \eng{UTF-8} βασίζεται στο πρότυπο
\eng{Unicode\footnote{https://home.unicode.org/}} το οποίο προσφέρει
143,859 χαρακτήρες. Το βασικότερο πλεονέκτημα της \eng{UTF-8} είναι η ευρύτατη
υποστήριξη που έχει παγκοσμίως, διευκολύνοντας τη διασυνδεσιμότητα στον
Εννοιολογικό Ιστό. Τα \eng{URIs} (βλ. και Παρατήρηση \ref{URIRemark}) επιτρέπουν
να περιγράφουμε με μοναδικό τρόπο οτιδήποτε. Έτσι γίνεται ευκολότερος ο συνδυασμός
δεδομένων από διαφορετικές πηγές.

Οι βασικοί κανόνες σύνταξης προκύπτουν από την γλώσσα σήμανσης
\eng{XML (Extensible Markup Language)\footnote{https://www.w3.org/TR/xml/}} και
περιλαμβάνουν:

\eng{\lstset{ basicstyle=\ttfamily\small, numbers=left}}

\begin{itemize}
	
	\item Ετικέτες (\eng{tags}) της μορφής \eng{\lstinline +<X>,</X>+}
	και \eng{\lstinline +<X/>+}

	\item Στοιχεία (\eng{elements}) που αποτελούνται από ετικέτες και
	κάποιο περιεχόμενο όπως \eng{\lstinline +<greeting>Hello, world!</greeting>+}
	ή ετικέτες χωρίς περιεχόμενο όπως \eng{\lstinline +<greet/>+}.

	\item Χαρακτηριστικά (\eng{attributes}) που συνοδεύουν μια ετικέτα
	όπως\\
	\eng{\lstinline[stringstyle=\ttfamily]{ <greeting wave=true>Hello!</greeting>}}
	ή \eng{\lstinline +<greet name=John/>+}

\end{itemize}

\subsection{Ανταλλαγή Δεδομένων και Ταξινομίες}

\quad Το μοντέλο \eng{RDF} επιτρέπει τον ορισμό αντικειμένων στον Παγκόσμιο
Ιστό καθώς και των σχέσεων και των ιδιοτήτων τους. Βασίζεται σε τριάδες της μορφής
``υποκείμενο - κατηγόρημα - αντικείμενο`` (\eng{subject - predicate - object}).
Συγκεκριμένα θα μελετήσουμε την παραλλαγή \eng{RDF/XML} η οποία βασίζεται στους
κανόνες σύνταξης της \eng{XML}. Επιπλέον αξιοποιεί τα \eng{URIs} για τις
ονομασίες αντικείμένων.

\begin{remark}
	Η μορφή \eng{RDF/XML} είναι η πρώτη που δημιουργήθηκε για το μοντέλο \eng{RDF}
	αλλά όχι η μοναδική. Σήμερα υπάρχουν διάφορες εναλλακτικές μορφές όπως:
	\begin{itemize}
		\eng{
		\item Turtle
		\item N-Triples
		\item N-Quads
		\item JSON-LD
		\item RDF/JSON
		}
	\end{itemize}
	Στόχος τους είναι μια αναπαράσταση πιο φιλική προς τον άνθρωπο, η μείωση των
	χαρακτήρων που απαιτούνται για μια πλήρη αναπαράσταση αλλά και οι δυνατότητες
	αποθήκευσης σε βάσεις ειδικά σχεδιασμένες για \eng{RDF} τριάδες
	(\eng{triplestores\footnote{https://en.wikipedia.org/wiki/Triplestore}})
\end{remark}

Στη συνέχεια παρουσιάζεται ένα απλό παράδειγμα που ξεκινά από μια δήλωση σε φυσική
γλώσσα, την τριάδα στην οποία αντιστοιχεί και την έκφρασή της σε \eng{RDF/XML}.

\begin{example}
	Έστω η ακόλουθη δήλωση σε φυσική γλώσσα (για ευκολία θα χρησιμοποιήσουμε
	την αγγλική γλώσσα):
	\begin{center}
		\eng{
		http://www.example.org/index.html
		has a creation date
		whose value is August 16, 1999.
		}
	\end{center}

	Αρχικά ορίζουμε κάποια προθέματα τα οποία μας επιτρέπουν να γράψουμε \eng{URIs}
	σε συντομευμένη μορφή και να καθορίσουμε τους τύπους δεδομένων που χρησιμοποιούνται.

	\eng{ \lstinputlisting{chapter2/URInamespaces} }

	Στη συνέχεια κατασκευάζουμε μια τριάδα από τα δεδομένα μας:

	\eng{
		\lstinline +ex:index.html  exterms:creation-date  "1999-08-16"\^\^xsd:date+
	}

	Το υποκείμενο (ιστοσελίδα) φέρει μια ιδιότητα (ημερομηνία δημιουργίας) με μια
	τιμή (1999-08-16). Επίσης ο τύπος αυτής της τιμής (ημερομηνία) καθορίζεται
	ξεχωριστά (\eng{\lstinline +\^\^xsd:date+}). Η τελική μορφή σε \eng{RDF/XML}
	είναι:

	\eng{ \lstinputlisting[language=XML]{chapter2/RDFexample.xml} }

	Tα βασικότερα σημεία της υλοποίησης είναι:

	\begin{itemize}
		\item Γραμμές 3 και 4 : Ορίζονται τα \eng{namespaces} που θα χρησιμοποιήσουμε.
		\item Γραμμή 5: Ξεκινά ο ορισμός μιας τριάδας με την ετικέτα\\
		\eng{\lstinline +<rdf:Description>+}
		\item Γραμμή 6: Ορίζεται το υποκείμενο της τριάδας με την ετικέτα\\
		\eng{\lstinline +<rdf:about>+}
		\item Γραμμή 7: Ξεκινά ο ορισμός του κατηγορήματος της τριάδας
		με την ετικέτα\\
		\eng{\lstinline +<exterms:creation-date>+}
		\item Γραμμή 8: Το χαρακτηριστικό \eng{\lstinline +<rdf:datatype>+}
		καθορίζει τον τύπο του στοιχείου εντός της ετικέτας.
	\end{itemize}
\end{example}

Η χρήση του μοντέλου \eng{RDF} μας επιτρέπει να ορίσουμε αντικείμενα, ιδιότητες και
σχέσεις. Αυτό όμως δεν είναι το ίδιο με το να αποτυπώσουμε σε αυτούς τους ορισμούς
κάποιο σημασιολογικό τους περιεχόμενο. Προκειμένου να επιτευχθεί κάτι τέτοιο,
χρησιμοποιούμε την γλώσσα \eng{RDFS}. Η γλώσσα αυτή εφοδιάζει το μοντέλο \eng{RDF}
με ένα λεξιλόγιο κατάλληλο για να ορίσουμε ταξινομίες, δηλαδή να ομαδοποιήσουμε
αντικείμενα με κοινά χαρακτηριστικά σε κατηγορίες καθώς και να ορίσουμε διάφορες
υποκατηγορίες.
